    const bookmarks = [

        description: 'Manage your team members by adding new users to your Go High Level account.'

      },

      {

        name: 'Main Dashboard',

        url: 'https://app.gohighlevel.com/v2/location/IYjyc7Efk0yAOzpSO9FM/dashboard',

        description: 'Access your main dashboard to monitor key metrics and activities.'

      },

      // Add more bookmarks here if needed

    ];


    // Function to render the bookmarks to the page

    function renderBookmarks() {

      const bookmarkListElement = document.getElementById('bookmark-list');

      bookmarkListElement.innerHTML = ''; // Clear the existing list

      bookmarks.forEach(bookmark => {

        const listItem = document.createElement('li');

        listItem.classList.add('bookmark-item');

        

        const link = document.createElement('a');

        link.href = bookmark.url;

        link.target = '_blank';

        link.textContent = bookmark.name;

        

        const description = document.createElement('p');

        description.textContent = bookmark.description;

        

        listItem.appendChild(link);

        listItem.appendChild(description);

        bookmarkListElement.appendChild(listItem);

      });

    }


    // Function to add a new bookmark

    function addBookmark(name, url, description) {

      bookmarks.push({ name, url, description });

      renderBookmarks(); // Re-render the bookmarks list to include the new one

    }


    // Initial render

    renderBookmarks();
